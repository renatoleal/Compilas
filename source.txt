% Exemplo de programa com declaração de funções auxiliares

function int maior ( int x , int y ) {
	int resposta := y ;
	if ( x > y ) then {
		resposta := x ;
	}
	return resposta ;
}

function int maior_de_tres ( int a , int b , int c ) {
	int r := 0 ;
	r := maior( a , r ) ;
	r := maior( b , r ) ;
	r := maior( c , r ) ;
	
	return r ;
}

main() {
	maior_de_tres( 1 , 2 , 3 ) ;
}
